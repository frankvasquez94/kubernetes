$sudo apt-get install nfs-kernel-server // Instalar el server nfs en todos los nodos
$sudo mkdir /storage // creat el storage en el node de controller
$sudo chmod 777 /storage // Asignarle permisos al storage
$sudo sh -c 'echo "/storage *(rw,no_root_squash)" > /etc/exports'  // exportar el stoage
$sudo systemctl enable --now nfs-server  // Habilitar nfs-server
$systemctl status nfs-server  // verifcar el status del server nfs-server
$sudo showmount -e localhost // ver el storage
